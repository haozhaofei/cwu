<script>
	window.document.oncontextmenu=function(ent){
		//兼容IE和火狐浏览器
		var event=ent ||window.event;
		//获取X坐标的位置
		var x=event.clientX;
		//获取Y坐标的位置
		var y=event.clientY;
		var uid=document.getElementById('uid');
		uid.style.top=y+"px";
		uid.style.left=x+"px";
		uid.style.display="block";
		//屏蔽浏览器右键
		return false;

	}
</script>
